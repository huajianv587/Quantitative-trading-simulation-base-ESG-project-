```python
class ErrorCode(Enum):
    @classmethod
    def get_by_category(cls, category: str) -> List['ErrorCode']:
        """按类别获取错误码"""
        return [e for e in cls if e.category == category]
    
    @classmethod
    def get_retryable_errors(cls) -> List['ErrorCode']:
        """获取所有可重试错误"""
        return [e for e in cls if e.retryable]
    
    @classmethod
    def get_high_severity_errors(cls) -> List['ErrorCode']:
        """获取高严重程度错误"""
        return [e for e in cls if e.severity == "high"]
```

#### 3.2 日志采样和限流
**当前问题**：高频错误可能导致日志爆炸

**优化方案**：
```python
class SamplingLogger:
    def __init__(self, logger, sample_rate=0.1):
        self.logger = logger
        self.sample_rate = sample_rate
        self.error_counts = {}
    
    def error(self, msg, **kwargs):
        error_key = hash(msg)
        count = self.error_counts.get(error_key, 0)
        
        # 前10次全记录，之后采样
        if count < 10 or random.random() < self.sample_rate:
            self.logger.error(msg, **kwargs)
        
        self.error_counts[error_key] = count + 1
```

**收益**：
- 防止日志爆炸
- 降低存储成本
- 保持关键信息

```python
class ErrorAggregator:
    def __init__(self, window_seconds=60):
        self.window = window_seconds
        self.errors = defaultdict(list)
    
    def should_alert(self, error_code: str) -> bool:
        now = time.time()
        recent = [t for t in self.errors[error_code] 
                  if now - t < self.window]
        self.errors[error_code] = recent + [now]
        
        # 1分钟内超过10次才告警
```

### 五、前端错误处理增强

#### 5.1 错误重试策略优化
**当前问题**：重试策略过于简单

**优化方案**：
```javascript
class RetryStrategy {
  constructor(maxRetries = 3, baseDelay = 1000) {
    this.maxRetries = maxRetries;
    this.baseDelay = baseDelay;
  }
  
  async execute(fn, errorCode) {
    for (let i = 0; i < this.maxRetries; i++) {
      try {
        return await fn();
      } catch (error) {
        if (!this.shouldRetry(error, i)) throw error;
        await this.delay(i, errorCode);
      }
    }
  }
  
  shouldRetry(error, attempt) {
    // 根据错误码决定是否重试
    if (!error.retryable) return false;
    if (attempt >= this.maxRetries - 1) return false;
    
    // 特定错误码的重试策略
    if (error.code === 'ERR_EXT_004') {
      // 限流错误，延长重试间隔
      return true;
    }
    return true;
  }
  
  async delay(attempt, errorCode) {
    // 根据错误码调整延迟策略
    let delay = this.baseDelay * Math.pow(2, attempt);
    
    if (errorCode === 'ERR_EXT_004') {
      delay *= 2; // 限流错误加倍延迟
    }
    
    // 添加随机抖动
    delay += Math.random() * 1000;
    
    await new Promise(resolve => setTimeout(resolve, delay));
  }
}
```

```javascript
class ErrorReporter {
  constructor(endpoint = '/api/errors/report') {
    this.endpoint = endpoint;
    this.buffer = [];
    this.flushInterval = 10000; // 10秒
    this.startAutoFlush();
  }
  
  report(error) {
    this.buffer.push({
      code: error.code,
      message: error.message,
      trace_id: error.trace_id,
      url: window.location.href,
      user_agent: navigator.userAgent,
      timestamp: new Date().toISOString(),
      stack: error.stack
    });
    
    if (this.buffer.length >= 10) {
      this.flush();
    }
  }
  
  async flush() {
    if (this.buffer.length === 0) return;
    
    const errors = this.buffer.splice(0);
    try {
      await fetch(this.endpoint, {
        method: 'POST',
        body: JSON.stringify({ errors }),
        headers: { 'Content-Type': 'application/json' }
      });
    } catch (e) {
      console.error('Failed to report errors', e);
    }
  }
}
```
