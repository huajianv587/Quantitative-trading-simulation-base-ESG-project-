## 🔴 P2: 后端异步化 (待实施)

### 问题分析
**位置**: `/gateway/db/supabase_client.py`, `/gateway/api/routers/agent.py`

**现状**:
```python
# 当前 - 同步阻塞
def get_history(session_id: str, limit: int = 20) -> list[dict]:
    resp = get_client().table("chat_history").select(...).execute()
    return [{"role": r["role"], "content": r["content"]} for r in resp.data]
```

**问题**:
- 所有数据库操作都是同步的
- 高并发下会导致线程池耗尽
- 吞吐量受限

### 优化方案

```python
# 改进 - 异步非阻塞
import asyncio
from concurrent.futures import ThreadPoolExecutor

# 创建线程池
executor = ThreadPoolExecutor(max_workers=10)

async def get_history_async(session_id: str, limit: int = 20) -> list[dict]:
    loop = asyncio.get_event_loop()
    
    def _fetch():
        return get_client().table("chat_history")\
            .select("role, content, created_at")\
            .eq("session_id", session_id)\
            .order("created_at", desc=False)\
            .limit(limit)\
            .execute()
    
    resp = await loop.run_in_executor(executor, _fetch)
    return [{"role": r["role"], "content": r["content"]} for r in resp.data]

# 路由改为异步
@router.get("/api/history/{session_id}")
async def get_chat_history(session_id: str, limit: int = 20):
    messages = await get_history_async(session_id, limit)
    return {"messages": messages}
```

**预期收益**:
- 吞吐量提升 3-5倍
- 支持 100+ 并发用户
- 响应时间降低 50%

**工作量**: 3-5天

async def get_history_paginated(
    session_id: str, 
    limit: int = 20, 
    offset: int = 0
) -> dict:
    # 获取总数 (使用count)
    count_resp = get_client().table("chat_history")\
        .select("count", count="exact")\
        .eq("session_id", session_id)\
        .execute()

from fastapi import Depends, HTTPException, status

from jose import JWTError, jwt
from datetime import datetime, timedelta

SECRET_KEY = os.getenv("JWT_SECRET_KEY")

ACCESS_TOKEN_EXPIRE_MINUTES = 30

security = HTTPBearer()

def create_access_token(data: dict):
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES)
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

async def get_current_user(credentials: HTTPAuthorizationCredentials = Depends(security)):
    try:
        token = credentials.credentials
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        user_id: str = payload.get("sub")
        if user_id is None:
            raise HTTPException(status_code=401, detail="Invalid token")
        return user_id
    except JWTError:
        raise HTTPException(status_code=401, detail="Invalid token")

# 后端: FastAPI WebSocket
from fastapi import WebSocket, WebSocketDisconnect

class ConnectionManager:
    def __init__(self):
        self.active_connections: dict[str, WebSocket] = {}
    
    async def connect(self, execution_id: str, websocket: WebSocket):
        await websocket.accept()
        self.active_connections[execution_id] = websocket
    
    def disconnect(self, execution_id: str):
        self.active_connections.pop(execution_id, None)
    
    async def send_update(self, execution_id: str, data: dict):
        websocket = self.active_connections.get(execution_id)
        if websocket:
            await websocket.send_json(data)

manager = ConnectionManager()

@app.websocket("/ws/execution/{execution_id}")
async def websocket_execution(websocket: WebSocket, execution_id: str):
    await manager.connect(execution_id, websocket)
    try:
        while True:
            # 从数据库获取最新状态
            status = await get_execution_status(execution_id)
            await manager.send_update(execution_id, status)
            await asyncio.sleep(1)
    except WebSocketDisconnect:
        manager.disconnect(execution_id)
```

// 前端: WebSocket客户端
export function openExecutionWS(executionId, onUpdate) {
  const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
  const ws = new WebSocket(`${protocol}//${window.location.host}/ws/execution/${executionId}`);
  
  ws.onopen = () => {
    console.log(`[WebSocket] Connected to execution ${executionId}`);
  };
  
  ws.onmessage = (event) => {
    const status = JSON.parse(event.data);
    onUpdate(status);
  };
  
  ws.onerror = (error) => {
    console.error('[WebSocket] Error:', error);
  };
  
  ws.onclose = () => {
    console.log('[WebSocket] Disconnected');
  };
  
  return ws;
}
```

from prometheus_client import Counter, Histogram, generate_latest

request_count = Counter(
    'api_requests_total', 
    'Total API requests', 
    ['method', 'endpoint', 'status']
)

request_duration = Histogram(
    'api_request_duration_seconds', 
    'API request duration', 
    ['endpoint']
)

@app.middleware("http")
async def add_metrics(request: Request, call_next):
    start = time.time()
    response = await call_next(request)

        endpoint=request.url.path,
        status=response.status_code
